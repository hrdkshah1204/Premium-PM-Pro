//app.js
import { auth, db } from './db.js';
import { onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-auth.js";
import { doc, getDoc, setDoc } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js";

// Import your business logic modules
import { handleEmailSignIn, handleRegistration, handleGoogleLogin, handlePasswordReset, toggleAuthUI } from './auth.js';
import { saveNewProject, loadUserProjects, saveNewTeam, renderAnalytics } from './admin.js';

// --- INITIALIZATION & AUTH GATE ---
onAuthStateChanged(auth, async (user) => {
    if (user) {
        if (!user.emailVerified) {
            document.getElementById('auth-status').innerText = "Please verify your email before logging in!";
            signOut(auth);
            return;
        }
        
        const userSnap = await getDoc(doc(db, "users", user.email));
        if (userSnap.exists() && userSnap.data().onboardingComplete) {
            showView('admin-view');
            document.getElementById('display-workspace-name').innerText = userSnap.data().workspaceName;
            document.getElementById('display-user-email').innerText = user.email;
            loadUserProjects();
        } else {
            showView('onboarding-view');
        }
    } else {
        showView('login-view');
    }
});

// --- UI EVENT LISTENERS ---

// Auth Buttons
document.getElementById('btn-primary-auth').addEventListener('click', () => {
    const isReg = document.getElementById('auth-title').innerText.includes("Join");
    isReg ? handleRegistration() : handleEmailSignIn();
});

document.getElementById('btn-toggle-auth').addEventListener('click', toggleAuthUI);
document.getElementById('btn-forgot-pass').addEventListener('click', handlePasswordReset);
document.getElementById('btn-google-login').addEventListener('click', handleGoogleLogin);
document.getElementById('btn-logout').addEventListener('click', () => signOut(auth));

// Onboarding Button
document.getElementById('btn-complete-onboarding').addEventListener('click', async () => {
    await setDoc(doc(db, "users", auth.currentUser.email), {
        name: document.getElementById('ob-name').value,
        workspaceName: document.getElementById('ob-workspace-name').value,
        company: document.getElementById('ob-company').value,
        onboardingComplete: true,
        role: 'admin'
    }, { merge: true });
    location.reload();
});

// Admin Suite Buttons
document.getElementById('btn-save-project').addEventListener('click', saveNewProject);
document.getElementById('btn-save-team').addEventListener('click', saveNewTeam);

// Navigation
document.querySelectorAll('.sidebar-btn').forEach(btn => {
    btn.addEventListener('click', (e) => {
        const pageId = e.target.getAttribute('data-page');
        document.querySelectorAll('.sub-page').forEach(p => p.classList.add('hidden'));
        document.getElementById(pageId).classList.remove('hidden');
        
        document.querySelectorAll('.sidebar-btn').forEach(b => b.classList.remove('active'));
        e.target.classList.add('active');

        if (pageId === 'sub1') loadUserProjects();
        if (pageId === 'sub6') renderAnalytics();
    });
});

// Helper function to switch main views
function showView(viewId) {
    ['login-view', 'onboarding-view', 'admin-view'].forEach(id => {
        document.getElementById(id).classList.add('hidden');
    });
    document.getElementById(viewId).classList.remove('hidden');
}
